・すべてのテーブルを結合して表示
SELECT * FROM sales s 
LEFT JOIN accounts a ON s.account_id = a.account_id
LEFT JOIN categories c ON s.category_id = c.category_id;

・特定のデータを取り出す
SELECT * FROM sales s 
LEFT JOIN accounts a ON s.account_id = a.account_id
LEFT JOIN categories c ON s.category_id = c.category_id
WHERE sale_id = 1;


・検索条件
select * from sales s
LEFT JOIN accounts a ON s.account_id = a.account_id
LEFT JOIN categories c ON s.category_id = c.category_id
where sale_date >= '' 
and sale_date <= ''
and account_id = ''
and category_id = ''
